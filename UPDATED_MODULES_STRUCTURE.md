## 🏗️ **STRUKTUR DATABASE MULTI-LEMBAGA**

### **Core Tables:**
```sql
-- Lembaga utama
CREATE TABLE lembaga (
    id SERIAL PRIMARY KEY,
    kode VARCHAR(10) UNIQUE, -- MTS, SMP, MA, TAHFIDZ
    nama VARCHAR(100),
    jenis VARCHAR(20), -- FORMAL, DINIAH, TAHFIDZ
    kementerian VARCHAR(50), -- KEMENAG, KEMENDIKBUD
    kepala_lembaga_id INTEGER REFERENCES users(id),
    is_active BOOLEAN DEFAULT true
);

-- Santri dengan mapping multi-lembaga
CREATE TABLE santri (
    id SERIAL PRIMARY KEY,
    nis VARCHAR(20) UNIQUE,
    nama_lengkap VARCHAR(200),
    -- Data pribadi...
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Mapping santri ke lembaga (bisa multiple)
CREATE TABLE santri_lembaga (
    santri_id INTEGER REFERENCES santri(id),
    lembaga_id INTEGER REFERENCES lembaga(id),
    tahun_ajaran VARCHAR(9), -- 2024/2025
    kelas VARCHAR(20),
    status VARCHAR(20) DEFAULT 'aktif',
    PRIMARY KEY (santri_id, lembaga_id, tahun_ajaran)
);

-- Modul Tahfidz Khusus
CREATE TABLE tahfidz_target (
    id SERIAL PRIMARY KEY,
    santri_id INTEGER REFERENCES santri(id),
    target_juz INTEGER, -- 1-30
    target_halaman INTEGER, -- 1-604
    tanggal_mulai DATE,
    tanggal_target DATE,
    status VARCHAR(20) DEFAULT 'active'
);

CREATE TABLE tahfidz_setoran (
    id SERIAL PRIMARY KEY,
    santri_id INTEGER REFERENCES santri(id),
    tanggal DATE,
    juz INTEGER,
    halaman_mulai INTEGER,
    halaman_selesai INTEGER,
    metode VARCHAR(20), -- SETORAN, MUROJAAH
    nilai_tajwid INTEGER, -- 1-100
    nilai_kelancaran INTEGER, -- 1-100
    catatan TEXT,
    disetorkan_ke INTEGER REFERENCES users(id)
);
```
